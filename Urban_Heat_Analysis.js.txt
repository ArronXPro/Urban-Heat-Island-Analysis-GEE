// ====== CONFIG ======
const JAMENDO_CLIENT_ID = "YOUR_JAMENDO_CLIENT_ID"; // <-- Paste your key here
const API = "https://api.jamendo.com/v3.0/tracks";

// ====== STATE ======
let tracks = [];
let currentIndex = -1;
let sound = null;
let favorites = JSON.parse(localStorage.getItem("favorites") || "[]");
let queue = [];
let view = "home";

// ====== ELEMENTS ======
const trackGrid = document.getElementById("trackGrid");
const searchInput = document.getElementById("searchInput");
const searchBtn = document.getElementById("searchBtn");
const playBtn = document.getElementById("playBtn");
const prevBtn = document.getElementById("prevBtn");
const nextBtn = document.getElementById("nextBtn");
const seek = document.getElementById("seek");
const volume = document.getElementById("volume");
const nowImg = document.getElementById("nowImg");
const nowTitle = document.getElementById("nowTitle");
const nowArtist = document.getElementById("nowArtist");
const curTime = document.getElementById("curTime");
const durTime = document.getElementById("durTime");
const favBtn = document.getElementById("favBtn");
const downloadBtn = document.getElementById("downloadBtn");
const sectionTitle = document.getElementById("sectionTitle");
const themeColor = document.getElementById("themeColor");

// ====== THEME ======
function setTheme(color) {
  document.documentElement.style.setProperty("--accent", color);
  localStorage.setItem("accent", color);
}
themeColor.addEventListener("input", e => setTheme(e.target.value));
document.querySelectorAll(".preset-colors span").forEach(el => {
  el.addEventListener("click", () => {
    const c = el.dataset.color;
    themeColor.value = c;
    setTheme(c);
  });
});
const savedColor = localStorage.getItem("accent");
if (savedColor) { themeColor.value = savedColor; setTheme(savedColor); }

// ====== NAVIGATION ======
document.querySelectorAll(".nav-btn").forEach(btn => {
  btn.addEventListener("click", () => {
    document.querySelectorAll(".nav-btn").forEach(b => b.classList.remove("active"));
    btn.classList.add("active");
    view = btn.dataset.view;
    if (view === "home") loadTrending();
    if (view === "favorites") showFavorites();
    if (view === "queue") showQueue();
  });
});

// ====== FETCH TRACKS ======
async function loadTrending() {
  sectionTitle.textContent = "🔥 Trending Now";
  const url = `${API}/?client_id=${JAMENDO_CLIENT_ID}&format=json&limit=24&order=popularity_total`;
  const res = await fetch(url);
  const data = await res.json();
  tracks = data.results;
  renderTracks(tracks);
}

async function searchTracks(q) {
  sectionTitle.textContent = `🔍 Results for "${q}"`;
  const url = `${API}/?client_id=${JAMENDO_CLIENT_ID}&format=json&limit=30&search=${encodeURIComponent(q)}`;
  const res = await fetch(url);
  const data = await res.json();
  tracks = data.results;
  renderTracks(tracks);
}

searchBtn.addEventListener("click", () => {
  const q = searchInput.value.trim();
  if (q) searchTracks(q);
});
searchInput.addEventListener("keypress", e => {
  if (e.key === "Enter") searchBtn.click();
});

// ====== RENDER ======
function renderTracks(list) {
  trackGrid.innerHTML = "";
  if (!list.length) {
    trackGrid.innerHTML = `<p style="color:var(--muted)">No tracks found.</p>`;
    return;
  }
  list.forEach((t, i) => {
    const card = document.createElement("div");
    card.className = "track-card";
    card.innerHTML = `
      <img src="${t.image}" alt="" />
      <div class="title">${t.name}</div>
      <div class="artist">${t.artist_name}</div>
    `;
    card.addEventListener("click", () => playTrack(i, list));
    // Preload audio for instant play
    const preload = new Audio();
    preload.preload = "auto";
    preload.src = t.audio;
    trackGrid.appendChild(card);
  });
}

function showFavorites() {
  sectionTitle.textContent = "❤️ Your Favorites";
  tracks = favorites;
  renderTracks(favorites);
}

function showQueue() {
  sectionTitle.textContent = "🎵 Play Queue";
  renderTracks(queue);
}

// ====== PLAYER ======
function playTrack(index, list) {
  if (sound) sound.unload();
  currentIndex = index;
  tracks = list;
  const t = list[index];

  sound = new Howl({
    src: [t.audio],
    html5: true,
    volume: parseFloat(volume.value),
    onplay: () => {
      playBtn.textContent = "⏸";
      durTime.textContent = formatTime(sound.duration());
      requestAnimationFrame(updateSeek);
    },
    onpause: () => playBtn.textContent = "▶",
    onend: () => nextBtn.click(),
  });
  sound.play();

  nowImg.src = t.image;
  nowTitle.textContent = t.name;
  nowArtist.textContent = t.artist_name;
  updateFavBtn();
}

function updateSeek() {
  if (sound && sound.playing()) {
    const s = sound.seek();
    seek.value = (s / sound.duration()) * 100;
    curTime.textContent = formatTime(s);
    requestAnimationFrame(updateSeek);
  }
}

seek.addEventListener("input", () => {
  if (sound) sound.seek((seek.value / 100) * sound.duration());
});

playBtn.addEventListener("click", () => {
  if (!sound) return;
  sound.playing() ? sound.pause() : sound.play();
});

prevBtn.addEventListener("click", () => {
  if (currentIndex > 0) playTrack(currentIndex - 1, tracks);
});
nextBtn.addEventListener("click", () => {
  if (currentIndex < tracks.length - 1) playTrack(currentIndex + 1, tracks);
});

volume.addEventListener("input", () => {
  if (sound) sound.volume(parseFloat(volume.value));
});

function formatTime(s) {
  const m = Math.floor(s / 60);
  const sec = Math.floor(s % 60).toString().padStart(2, "0");
  return `${m}:${sec}`;
}

// ====== FAVORITES ======
favBtn.addEventListener("click", () => {
  if (currentIndex < 0) return;
  const t = tracks[currentIndex];
  const idx = favorites.findIndex(f => f.id === t.id);
  if (idx >= 0) favorites.splice(idx, 1);
  else favorites.push(t);
  localStorage.setItem("favorites", JSON.stringify(favorites));
  updateFavBtn();
});

function updateFavBtn() {
  if (currentIndex < 0) return;
  const t = tracks[currentIndex];
  const isFav = favorites.some(f => f.id === t.id);
  favBtn.textContent = isFav ? "❤️" : "🤍";
}

// ====== DOWNLOAD (Jamendo tracks are legally downloadable) ======
downloadBtn.addEventListener("click", () => {
  if (currentIndex < 0) return;
  const t = tracks[currentIndex];
  const a = document.createElement("a");
  a.href = t.audiodownload || t.audio;
  a.download = `${t.artist_name} - ${t.name}.mp3`;
  a.click();
});
